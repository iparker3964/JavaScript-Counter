                    //get the values from the page
                    //starts or controller function
                    function getValues(){
                        //get values from the page
                        let startVal = document.getElementById("startValue").value;
                        let endVal = document.getElementById("endValue").value;
                        let numbers = [];
                    
                        //validate our input
                        //parse into int
                        startVal = parseInt(startVal);
                        endVal = parseInt(endVal);
                        //we call generateNums
                        if(Number.isInteger(startVal) && Number.isInteger(endVal)){
                            numbers = generateNums(startVal,endVal);
                    
                        }else{
                            alert("You must enter integers!");
                        }
                        
                        //we call displayNumbers
                        if(numbers.length > 0){
                            displayNumbers(numbers);
                        }
                    }
                    //generate numbers from the startvalue to the end value
                    //logic function(s)
                    function generateNums(start,end){
                        let nums = [];
                        for(let i = start; i <= end; i++){
                            nums.push(i);
                        }
                        return nums;
                    }
                    //display the numbers and mark even numbers bold
                    //display or view function
                    function displayNumbers(numsList){
                        let tempRows = "";
                    
                        for(let i = 0; i < numsList.length; i++){
                            let num = numsList[i];
                    
                            let className = "even";
                    
                            if(num % 2 == 0){
                                className = "even";
                            }
                            else{
                                className = "odd"
                            }
                            //This does render correctly with Prism see the source code.
                            tempRows += `${num}`;
   
                        }
                        document.getElementById("results").innerHTML = tempRows;
                    }

The Code is structured in three functions.

getValues

getValues accepts the user input from the page. It utilizes getElementById to pull the values from the page. It passes those values to the generateNums() function. The function generateNums() returns an array of values and passes that array to the displayNumbers() function.

generateNums

generateNums takes in two parameters startValue and endValue. We create a variable(nums) that holds an array. A for loop is used to generate all of the numbers between startValue and endValue.

displayNumbers

displayNumbers takes in two parameters startValue and endValue. We create a variable(nums) that holds an array. A for loop is used to generate all of the numbers between startValue and endValue. We create a variable (className) that holds the name of a CSS class that we will use to bold the numbers. We create a variable tempRows that will hold the html we will write to the page. A for loop is used to check all of the numbers to see if they are even or odd. The remainder operator(%) used to see if the number is divisable by two. If it returns zero then it is even. If not its odd. The correct className is injected into the html row for display. Each iteration of the loop adds to the tempRows variable. At the end of the loop the resulting HTML rows are written to the page.
